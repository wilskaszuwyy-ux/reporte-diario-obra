Vincular Google Sheets

Los datos se envían de forma directa a tu Google Sheets a través de una macro en Apps Script.

Pega aquí la URL de tu script ejecutado como aplicación web.

Instrucciones del Servidor Google Sheets:

  1. Crea una hoja de cálculo en Google Sheets.
  2. Ve a Extensiones > Apps Script.
  3. Borra el código existente y pega el siguiente script:
function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); if (sheet.getLastRow() === 0) { sheet.appendRow([ "Fecha", "Responsable", "Proyecto", "Clima", "Capataces", "Operarios", "Oficiales", "Peones", "Subcontratos", "Jornada Laboral", "Tareas del Día", "Avance Global %", "Materiales Recibidos", "Incidentes", "Observaciones", "Firma" ]); sheet.getRange(1, 1, 1, 16).setFontWeight("bold").setBackground("#F1F5F9"); } var tareas = data.tareas.map(function(t) { return t.tarea + " (" + t.avance + "%)"; }).join("\n"); var materiales = data.materiales.map(function(m) { return m.cant + " " + m.un + " de " + m.material; }).join("\n"); sheet.appendRow([ data.fecha, data.responsable, data.proyecto, data.clima, data.personal.capataz, data.personal.operario, data.personal.oficial, data.personal.peon, data.personal.subcontrato, data.jornada.inicio + " a " + data.jornada.fin, tareas, data.avanceGlobal + "%", materiales, data.incidentes, data.observaciones, data.firma ]); sheet.autoResizeColumns(1, 16); return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
  1. Haz clic en Implementar > Nueva implementación.
  2. Elige tipo: Aplicación web.
  3. Ejecutar como: Tú. Acceso: Cualquiera.
  4. Haz clic en Implementar, copia la URL y pégala arriba o configúrala en el código como constante.
